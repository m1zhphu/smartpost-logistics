-- ==============================================================================
-- HỆ THỐNG QUẢN LÝ VẬN HÀNH LOGISTICS SMARTPOST - SCRIPT TẠO DATABASE
-- ==============================================================================

-- ==========================================
-- PHẦN 1: DỮ LIỆU NỀN TẢNG (MASTER DATA)
-- ==========================================

CREATE TABLE departments (
    department_id SERIAL PRIMARY KEY,
    department_code VARCHAR(50) UNIQUE NOT NULL,
    department_name VARCHAR(100) NOT NULL,
    status BOOLEAN DEFAULT TRUE
);

CREATE TABLE roles (
    role_id SERIAL PRIMARY KEY,
    role_name VARCHAR(100) NOT NULL,
    permissions JSONB, -- Sử dụng JSONB của Postgres để tối ưu hiệu năng
    is_active BOOLEAN DEFAULT TRUE
);

CREATE TABLE vendors (
    vendor_id SERIAL PRIMARY KEY,
    vendor_code VARCHAR(50) UNIQUE NOT NULL,
    vendor_name VARCHAR(255) NOT NULL,
    service_type VARCHAR(50),
    status BOOLEAN DEFAULT TRUE
);

CREATE TABLE pricing_policies (
    policy_id SERIAL PRIMARY KEY,
    policy_code VARCHAR(50) UNIQUE NOT NULL,
    policy_name VARCHAR(255) NOT NULL,
    policy_type VARCHAR(20) NOT NULL,
    valid_from TIMESTAMP,
    valid_to TIMESTAMP,
    is_approved BOOLEAN DEFAULT FALSE,
    is_active BOOLEAN DEFAULT TRUE
);

-- TẠO BẢNG HUBS (Chưa gắn FK manager_id để tránh vòng lặp với Users)
CREATE TABLE hubs (
    hub_id SERIAL PRIMARY KEY,
    hub_code VARCHAR(20) UNIQUE NOT NULL,
    hub_name VARCHAR(150) NOT NULL,
    hub_type VARCHAR(50),
    province_id INT,
    address_detail VARCHAR(255),
    manager_id INT, -- Sẽ được ALTER thêm Khóa ngoại sau khi tạo bảng Users
    status BOOLEAN DEFAULT TRUE
);

CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    phone_number VARCHAR(20),
    role_id INT REFERENCES roles(role_id),
    department_id INT REFERENCES departments(department_id),
    primary_hub_id INT REFERENCES hubs(hub_id),
    vehicle_plate VARCHAR(50),
    status BOOLEAN DEFAULT TRUE
);

-- Cập nhật lại Khóa ngoại cho bảng Hubs
ALTER TABLE hubs ADD CONSTRAINT fk_hubs_manager FOREIGN KEY (manager_id) REFERENCES users(user_id);

CREATE TABLE user_data_access (
    access_id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(user_id),
    accessible_hub_id INT REFERENCES hubs(hub_id)
);

CREATE TABLE customers (
    customer_id SERIAL PRIMARY KEY,
    customer_code VARCHAR(50) UNIQUE NOT NULL,
    customer_type VARCHAR(20) NOT NULL,
    tax_code VARCHAR(50),
    company_name VARCHAR(255),
    transaction_name VARCHAR(255),
    identity_no VARCHAR(50),
    representative_name VARCHAR(100),
    province_id INT, -- Tạm dùng INT do chưa có bảng Hành chính
    district_id INT,
    ward_id INT,
    address_detail VARCHAR(255),
    parent_customer_id INT REFERENCES customers(customer_id),
    staff_in_charge_id INT REFERENCES users(user_id),
    status VARCHAR(20) DEFAULT 'ACTIVE'
);

CREATE TABLE customer_price_mapping (
    mapping_id SERIAL PRIMARY KEY,
    customer_id INT REFERENCES customers(customer_id),
    policy_id INT REFERENCES pricing_policies(policy_id)
);

CREATE TABLE bank_accounts (
    account_id SERIAL PRIMARY KEY,
    customer_id INT UNIQUE REFERENCES customers(customer_id),
    bank_name VARCHAR(255) NOT NULL,
    account_number VARCHAR(50) NOT NULL,
    account_name VARCHAR(100) NOT NULL,
    is_verified BOOLEAN DEFAULT FALSE
);

-- ==========================================
-- PHẦN 2: TIẾP NHẬN YÊU CẦU (BOOKING)
-- ==========================================

CREATE TABLE booking_requests (
    request_id SERIAL PRIMARY KEY,
    request_code VARCHAR(50) UNIQUE NOT NULL,
    source VARCHAR(50),
    shop_order_code VARCHAR(100),
    customer_id INT REFERENCES customers(customer_id),
    sender_phone VARCHAR(20),
    pickup_address VARCHAR(255),
    target_hub_id INT REFERENCES hubs(hub_id),
    product_type VARCHAR(50),
    est_weight DECIMAL(10,2),
    is_vehicle_required BOOLEAN DEFAULT FALSE,
    status VARCHAR(50),
    assigned_shipper_id INT REFERENCES users(user_id)
);

-- ==========================================
-- PHẦN 3: QUẢN LÝ VẬN ĐƠN (WAYBILLS)
-- ==========================================

CREATE TABLE waybills (
    waybill_id SERIAL PRIMARY KEY,
    waybill_code VARCHAR(50) UNIQUE NOT NULL,
    request_id INT REFERENCES booking_requests(request_id),
    customer_id INT REFERENCES customers(customer_id),
    receiver_name VARCHAR(100),
    receiver_phone VARCHAR(20),
    receiver_address VARCHAR(255),
    origin_hub_id INT REFERENCES hubs(hub_id),
    dest_hub_id INT REFERENCES hubs(hub_id),
    service_type VARCHAR(50),
    delivery_type VARCHAR(50),
    actual_weight DECIMAL(10,2),
    converted_weight DECIMAL(10,2),
    payment_method VARCHAR(50),
    cod_amount DECIMAL(15,2) DEFAULT 0,
    shipping_fee DECIMAL(15,2) DEFAULT 0,
    extra_services_fee DECIMAL(15,2) DEFAULT 0,
    vat_amount DECIMAL(15,2) DEFAULT 0,
    total_amount_to_collect DECIMAL(15,2) DEFAULT 0,
    status VARCHAR(50),
    version INT DEFAULT 1 -- Khóa Optimistic Locking đã thống nhất từ MVP
);

CREATE TABLE waybill_items (
    item_id SERIAL PRIMARY KEY,
    waybill_id INT REFERENCES waybills(waybill_id),
    parcel_code VARCHAR(50) UNIQUE NOT NULL,
    product_group VARCHAR(100),
    actual_weight DECIMAL(10,2),
    converted_weight DECIMAL(10,2),
    length DECIMAL(10,2),
    width DECIMAL(10,2),
    height DECIMAL(10,2),
    quantity INT DEFAULT 1
);

CREATE TABLE waybill_extra_services (
    id SERIAL PRIMARY KEY,
    waybill_id INT REFERENCES waybills(waybill_id),
    service_name VARCHAR(100),
    service_fee DECIMAL(15,2) DEFAULT 0
);

-- ==========================================
-- PHẦN 4: VẬN HÀNH KHO (WAREHOUSE)
-- ==========================================

CREATE TABLE bags (
    bag_id SERIAL PRIMARY KEY,
    bag_code VARCHAR(50) UNIQUE NOT NULL,
    seal_number VARCHAR(50),
    dest_hub_id INT REFERENCES hubs(hub_id),
    total_weight DECIMAL(10,2),
    created_by INT REFERENCES users(user_id),
    status VARCHAR(50)
);

CREATE TABLE bag_items (
    bag_item_id SERIAL PRIMARY KEY,
    bag_id INT REFERENCES bags(bag_id),
    waybill_id INT REFERENCES waybills(waybill_id)
);

CREATE TABLE manifests (
    manifest_id SERIAL PRIMARY KEY,
    manifest_code VARCHAR(50) UNIQUE NOT NULL,
    manifest_type VARCHAR(50),
    from_hub_id INT REFERENCES hubs(hub_id),
    to_hub_id INT REFERENCES hubs(hub_id),
    vehicle_number VARCHAR(50),
    vendor_id INT REFERENCES vendors(vendor_id),
    shipper_id INT REFERENCES users(user_id),
    total_weight DECIMAL(10,2),
    total_items INT,
    dispatched_at TIMESTAMP
);

CREATE TABLE manifest_details (
    detail_id SERIAL PRIMARY KEY,
    manifest_id INT REFERENCES manifests(manifest_id),
    waybill_id INT REFERENCES waybills(waybill_id),
    bag_id INT REFERENCES bags(bag_id)
);

CREATE TABLE mail_trips (
    id BIGSERIAL PRIMARY KEY,
    trip_code VARCHAR(50) UNIQUE NOT NULL,
    transport_method VARCHAR(50),
    vehicle_plate VARCHAR(20),
    driver_phone VARCHAR(20),
    departure_time TIMESTAMP,
    status VARCHAR(20)
);

-- TẠO BẢNG TRACKING LOGS CÓ PHÂN VÙNG (PARTITION)
CREATE TABLE tracking_logs (
    id BIGSERIAL,
    waybill_id BIGINT, -- Giữ BIGINT để scale
    status_id VARCHAR(100), -- Đổi thành VARCHAR vì dữ liệu mẫu là "Phát thành công..."
    user_id INT,
    hub_id INT,
    manifest_id BIGINT,
    action_time TIMESTAMP,
    system_time TIMESTAMP NOT NULL,
    PRIMARY KEY (id, system_time) -- Cột phân vùng bắt buộc phải nằm trong Khóa chính
) PARTITION BY RANGE (system_time);

-- Tạo sẵn partition cho tháng hiện tại (Có thể viết hàm tự động tạo sau)
CREATE TABLE tracking_logs_current PARTITION OF tracking_logs
    FOR VALUES FROM ('2026-04-01 00:00:00') TO ('2026-05-01 00:00:00');

-- ==========================================
-- PHẦN 5: KẾ TOÁN & ĐỐI SOÁT (ACCOUNTING)
-- ==========================================

CREATE TABLE statement_cod (
    statement_id SERIAL PRIMARY KEY,
    statement_code VARCHAR(50) UNIQUE NOT NULL,
    customer_id INT REFERENCES customers(customer_id),
    total_bills INT,
    total_cod_amount DECIMAL(15,2),
    created_at TIMESTAMP,
    status VARCHAR(50)
);

CREATE TABLE statement_debt (
    statement_id SERIAL PRIMARY KEY,
    statement_code VARCHAR(50) UNIQUE NOT NULL,
    customer_id INT REFERENCES customers(customer_id),
    total_main_fee DECIMAL(15,2),
    total_extra_fee DECIMAL(15,2),
    total_vat DECIMAL(15,2),
    grand_total DECIMAL(15,2)
);

CREATE TABLE statement_details (
    mapping_id SERIAL PRIMARY KEY,
    statement_id INT NOT NULL, -- Không dùng FK cứng vì nó trỏ về 2 bảng (COD hoặc Debt) tùy điều kiện
    waybill_id INT REFERENCES waybills(waybill_id),
    statement_type VARCHAR(20) -- Lọc loại 'COD' hay 'DEBT'
);

CREATE TABLE delivery_results (
    delivery_id SERIAL PRIMARY KEY,
    waybill_id INT REFERENCES waybills(waybill_id),
    shipper_id INT REFERENCES users(user_id),
    actual_cod_collected DECIMAL(15,2),
    pod_image_url VARCHAR(255),
    delivery_time TIMESTAMP,
    status VARCHAR(50)
);

CREATE TABLE payment_transactions (
    transaction_id SERIAL PRIMARY KEY,
    statement_id INT, -- Trỏ về bảng kê tùy loại
    amount_paid DECIMAL(15,2),
    bank_ref_code VARCHAR(100),
    payment_date TIMESTAMP
);

CREATE TABLE pods (
    id BIGSERIAL PRIMARY KEY,
    pod_code VARCHAR(50) UNIQUE NOT NULL,
    shipper_id INT REFERENCES users(user_id),
    created_by INT REFERENCES users(user_id),
    total_waybills INT,
    status VARCHAR(20),
    returned_time TIMESTAMP
);

-- ==========================================
-- PHẦN 6: XỬ LÝ SỰ CỐ & HÀNG HOÀN
-- ==========================================

CREATE TABLE incidents (
    incident_id SERIAL PRIMARY KEY,
    reference_type VARCHAR(20),
    reference_code VARCHAR(50),
    incident_category VARCHAR(100),
    hub_id INT REFERENCES hubs(hub_id),
    created_by INT REFERENCES users(user_id),
    surcharge_amount DECIMAL(15,2),
    incident_image_url VARCHAR(255),
    action_required VARCHAR(50),
    status VARCHAR(50)
);

CREATE TABLE incident_logs (
    log_id SERIAL PRIMARY KEY,
    incident_id INT REFERENCES incidents(incident_id),
    action_by INT REFERENCES users(user_id),
    action_detail TEXT,
    created_at TIMESTAMP
);

CREATE TABLE support_tickets (
    id BIGSERIAL PRIMARY KEY,
    ticket_code VARCHAR(50) UNIQUE NOT NULL,
    customer_id BIGINT, 
    waybill_id BIGINT,
    issue_category VARCHAR(50),
    content TEXT,
    status VARCHAR(20),
    assigned_to INT REFERENCES users(user_id)
);

-- ==========================================
-- PHẦN 7: CẤU HÌNH HỆ THỐNG
-- ==========================================

CREATE TABLE system_configs (
    config_key VARCHAR(100) PRIMARY KEY,
    config_value VARCHAR(255),
    description TEXT
);
